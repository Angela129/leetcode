# 455分饼干
```
/**
 * @param {number[]} g
 * @param {number[]} s
 * @return {number}
 */
var findContentChildren = function(g, s) {
    let result=0
    //先给两个数组排序
    g.sort((a,b)=>a-b)
    s.sort((a,b)=>a-b)
    //获得最大临界值
    let gi=g.length-1
    let si=s.length-1
    //把最大的饼干分给胃口最好的那个小孩
    while(gi>=0&&si>=0){
    if(g[gi]<=s[si]){
        result++
        si--
        gi--
    }else{
        //小孩不满足
        gi--
    }
    }
    return result
};
```
